17 def getBuiltinConfigs(self):
18 '''
19 Get the OCIO built in configurations.
20 Returnes a dictionary of color spaces and the default `ACES Cg Config`.
21 '''
22
23 # As of version 2.2, `ACES Cg Config` and `ACES Studio Config` are packaged with `OCIO`, meaning that
24 # they are available to use without having to download them separately. The `getBuiltinConfigs()`
25 # API is explained [here](https://opencolorio.readthedocs.io/en/latest/releases/ocio_2_2.html)
26
27 # Get the OCIO built in configs
28 registry = OCIO.BuiltinConfigRegistry().getBuiltinConfigs()
29
30 # Create a dictionary of configs
31 configs = {}
32 for item in registry:
33 # The short_name is the URI-style name.
34 # The ui_name is the name to use in a user interface.
35 short_name, ui_name, isRecommended, isDefault = item
36
37 # Don't present built-in configs to users if they are no longer recommended.
38 if isRecommended:
39 # Create a config using the Cg config
40 config = OCIO.Config.CreateFromBuiltinConfig(short_name)
41 colorSpaces = None
42 if config:
43 colorSpaces = config.getColorSpaces()
44
45 if colorSpaces:
46 configs[short_name] = [config, colorSpaces]
47
48 acesCgConfigPath = 'ocio://cg-config-v1.0.0_aces-v1.3_ocio-v2.1'
49 builtinCfgC = OCIO.Config.CreateFromFile(acesCgConfigPath)
50 print('Built-in config:', builtinCfgC.getName())
51 csnames = builtinCfgC.getColorSpaceNames()
52 print('- Number of color spaces: %d' % len(csnames))
53
54 return configs, builtinCfgC

88 def generateShaderCode(self, config, sourceColorSpace, destColorSpace, language):
89 '''
90 Generate shader for a transform from a source color space to a destination color space
91 for a given config and shader language.
92
93 Returns the shader code and the number of texture resources required.
94 '''
95 shaderCode = ''
96 textureCount = 0
97 if not config:
98 return shaderCode, textureCount
99
100 # Create a processor for a pair of colorspaces
101 processor = None
102 try:
103 processor = config.getProcessor(sourceColorSpace, destColorSpace)
104 except:
105 print('Failed to generated code for transform: %s -> %s' % (sourceColorSpace, destColorSpace))
106 return shaderCode, textureCount
107
108 if processor:
109 gpuProcessor = processor.getDefaultGPUProcessor()
110 if gpuProcessor:
111 shaderDesc = OCIO.GpuShaderDesc.CreateShaderDesc()
112 if shaderDesc:
113 try:
114 shaderDesc.setLanguage(language)
115 if shaderDesc.getLanguage() == language:
116 self.setShaderDescriptionParameters(shaderDesc, sourceColorSpace, destColorSpace, "color4")
117 gpuProcessor.extractGpuShaderInfo(shaderDesc)
118 shaderCode = shaderDesc.getShaderText()
119
120 for t in shaderDesc.getTextures():
121 textureCount += 1
122
123 if shaderCode:
124 shaderCode = shaderCode.replace(
125 "// Declaration of the OCIO shader function\n",
126 "// " + sourceColorSpace + " to " + destColorSpace + " function. Texture count: %d\n" % textureCount)
127
128 except OCIO.Exception as err:
129 print(err)
130
131 return shaderCode, textureCount

160 def hasTextureResources(self, configs, targetColorSpace, language):
161 '''
162 Scan through all the color spaces on the configs to check for texture resource usage.
163 Returns a set of color spaces that require texture resources.
164 '''
165 testedSources = set()
166 textureSources = set()
167 for c in configs:
168 config = OCIO.Config.CreateFromBuiltinConfig(c)
169 colorSpaces = config.getColorSpaces()
170 for colorSpace in colorSpaces:
171 colorSpaceName = colorSpace.getName()
172 # Skip if the colorspace is already tested
173 if colorSpaceName in testedSources:
174 continue
175 testedSources.add(colorSpaceName)
176
177 # Test for texture resource usage
178 code, textureCount = self.generateShaderCode(config, colorSpace.getName(), targetColorSpace, language)
179 if textureCount:
180 print('- Transform "%s" to "%s" requires %d texture resources' % (colorSpace.getName(), targetColorSpace, textureCount))
181 textureSources.add(colorSpaceName)
182
183 return textureSources

205 def generateMaterialXDefinition(self, doc, sourceColorSpace, targetColorSpace, inputName, type):
206 '''
207 Create a new definition in a document for a given color space transform.
208 Returns the definition.
209 '''
210 # Create a definition
211 transformName = self.createTransformName(sourceColorSpace, targetColorSpace, type)
212 nodeName = transformName.replace('mx_', 'ND_')
213
214 comment = doc.addChildOfCategory('comment')
215 docString = ' Color space %s to %s transform. Generated via OCIO. ' % (sourceColorSpace, targetColorSpace)
216 comment.setDocString(docString)
217
218 definition = doc.addNodeDef(nodeName, 'color4')
219 category = sourceColorSpace + '_to_' + targetColorSpace
220 definition.setNodeString(category)
221 definition.setNodeGroup('colortransform')
222 definition.setDocString(docString)
223 definition.setVersionString('1.0')
224
225 defaultValueString = '0.0 0.0 0.0 1.0'
226 defaultValue = mx.createValueFromStrings(defaultValueString, 'color4')
227 input = definition.addInput(inputName, type)
228 input.setValue(defaultValue)
229 output = definition.getOutput('out')
230 output.setAttribute('default', 'in')
231
232 return definition

245 def createMaterialXImplementation(self, sourceColorSpace, targetColorSpace, doc, definition, transformName, extension, target):
246 '''
247 Create a new implementation in a document for a given definition.
248 '''
249 implName = transformName + '_' + target
250 filename = transformName + '.' + extension
251 implName = implName.replace('mx_', 'IM_')
252
253 # Check if implementation already exists
254 impl = doc.getImplementation(implName)
255 if impl:
256 print('Implementation already exists: %s' % implName)
257 return impl
258
259 comment = doc.addChildOfCategory('comment')
260 comment.setDocString(' Color space %s to %s transform. Generated via OCIO for target: %s'
261 % (sourceColorSpace, targetColorSpace, target))
262 impl = doc.addImplementation(implName)
263 impl.setFile(filename)
264 impl.setFunction(transformName)
265 impl.setTarget(target)
266 impl.setNodeDef(definition)
267
268 return impl

270 def generateOCIO(self, config, definitionDoc, implDoc, sourceColorSpace = 'acescg', targetColorSpace = 'lin_rec709',
271 type='color4', IN_PIXEL_STRING = 'in'):
272 '''
273 Generate a MaterialX definition and implementation for a given color space transform.
274 Returns the definition, implementation, source code, extension and target.
275 '''
276
277 # List of MaterialX target language, source code extensions, and OCIO GPU languages
278 generationList = [
279 ['genglsl', 'glsl', OCIO.GpuLanguage.GPU_LANGUAGE_GLSL_4_0]
280 #, ['genmsl', 'metal', OCIO.GpuLanguage.GPU_LANGUAGE_MSL_2_0]
281 ]
282
283 definition = None
284 transformName = self.createTransformName(sourceColorSpace, targetColorSpace, type)
285 for gen in generationList:
286 target = gen[0]
287 extension = gen[1]
288 language = gen[2]
289
290 code, textureCount = self.generateShaderCode(config, sourceColorSpace, targetColorSpace, language)
291
292 # Skip if there are texture resources
293 if textureCount:
294 print('- Skip generation for transform: "%s" to "%s" which requires %d texture resources' % (sourceColorSpace, targetColorSpace, textureCount))
295 continue
296
297 if code:
298 # Create the definition once
299 if not definition:
300 # Create color4 variant
301 definition = self.generateMaterialXDefinition(definitionDoc, sourceColorSpace, targetColorSpace,
302 IN_PIXEL_STRING, type)
303 # Create color3 variant (nodegraph)
304 self.createColor3Variant(definition, definitionDoc, IN_PIXEL_STRING)
305
306 # Create the implementation
307 self.createMaterialXImplementation(sourceColorSpace, targetColorSpace, implDoc, definition, transformName, extension, target)
308
309 return definition, transformName, code, extension, target

311 def generateOCIOGraph(self, config, sourceColorSpace = 'acescg', targetColorSpace = 'lin_rec709',
312 type='color3'):
313 '''
314 Generate a MaterialX nodegraph for a given color space transform.
315 @param config: The OCIO configuration.
316 @param sourceColorSpace: The source color space.
317 @param targetColorSpace: The destination color space.
318 @param type: The type of the transform.
319 Returns a MaterialX document containing a functional nodegraph and nodedef pair.
320 '''
321 groupTransform = self.generateTransformGraph(config, sourceColorSpace, targetColorSpace)
322 result = f'{groupTransform}'
323
324 # To add. Proper testing of unsupported transforms...
325 invalidTransforms = [ OCIO.TransformType.TRANSFORM_TYPE_LUT3D, OCIO.TransformType.TRANSFORM_TYPE_LUT1D,
326 OCIO.TransformType.TRANSFORM_TYPE_RANGE,
327 OCIO.TransformType.TRANSFORM_TYPE_GRADING_PRIMARY ]
328
329 # Create a document, a nodedef and a functional graph.
330 graphDoc = mx.createDocument()
331 outputType = 'color3'
332 xformName = sourceColorSpace + '_to_' + targetColorSpace + '_' + outputType
333 nd = graphDoc.addNodeDef('ND_' + xformName )
334 nd.setAttribute('node', xformName)
335 ndInput = nd.addInput('in', 'color3')
336 ndInput.setValue(mx.createValueFromStrings('0.0 0.0 0.0', 'color3'))
337 ng = graphDoc.addNodeGraph('NG_' + xformName)
338 ng.setAttribute('nodedef', nd.getName())
339 convertNode = ng.addNode('convert', 'asVec', 'vector3')
340 converInput = convertNode.addInput('in', 'color3')
341 converInput.setInterfaceName('in')
342
343 #print(f'Transform from: {sourceColorSpace} to {targetColorSpace}')
344 if not groupTransform:
345 #print(f'No group transform found for the color space transform: {sourceColorSpace} to {targetColorSpace}')
346 return None
347 #print(f'Number of transforms: {groupTransform.__len__()}')
348 previousNode = None
349
350 # Iterate and create appropriate nodes and connections
351 for i in range(groupTransform.__len__()):
352 transform = groupTransform.__getitem__(i)
353 # Get type of transform
354 transformType = transform.getTransformType()
355 if transformType in invalidTransforms:
356 print(f'- Transform[{i}]: {transformType} contains an unsupported transform type')
357 continue
358
359 #print(f'- Transform[{i}]: {transformType}')
360 if transformType == OCIO.TransformType.TRANSFORM_TYPE_MATRIX:
361 matrixNode = ng.addNode('transform', ng.createValidChildName(f'matrixTransform'), 'vector3')
362
363 # Route output from previous node as input of current node
364 inInput = matrixNode.addInput('in', 'vector3')
365 if previousNode:
366 inInput.setAttribute('nodename', previousNode)
367 else:
368 if i==0:
369 inInput.setAttribute('nodename', 'asVec')
370 else:
371 inInput.setValue(mx.createValueFromStrings('0.0 0.0 0.0', 'vector3'))
372
373 # Set matrix value
374 matInput = matrixNode.addInput('mat', 'matrix33')
375 matrixValue = transform.getMatrix()
376 # Extract 3x3 matrix from 4x4 matrix
377 matrixValue = matrixValue[0:3] + matrixValue[4:7] + matrixValue[8:11]
378 matrixValue = ', '.join([str(x) for x in matrixValue])
379 #print(' - Matrix:', matrixValue)
380 matInput.setAttribute('value', matrixValue)
381
382 # Add offset value - TODO
383 offsetValue = transform.getOffset()
384 offsetValue = ', '.join([str(x) for x in offsetValue])
385 #print(' - Offset:', offsetValue)
386 # Add a add vector3 to graph
387
388 previousNode = matrixNode.getName()
389
390 # TODO: Handle other transform types
391 elif transformType == OCIO.TransformType.TRANSFORM_TYPE_EXPONENT_WITH_LINEAR:
392 print(f'- Transform[{i}]: {transformType} support has not been implemented yet')
393 elif transformType == OCIO.TransformType.TRANSFORM_TYPE_EXPONENT:
394 print(f'- Transform[{i}]: {transformType} support has not been implemented yet')
395 else:
396 print(f'- Transform[{i}]: {transformType} support has not been implemented yet')
397 continue
398
399
400 # Create an output for the last node if any
401 convertNode2 = ng.addNode('convert', 'asColor', 'color3')
402 converInput2 = convertNode2.addInput('in', 'vector3')
403 if previousNode:
404 converInput2.setAttribute('nodename', previousNode)
405 else:
406 # Pass-through
407 #print('No transforms applied. Transform is a pass-through.')
408 converInput2.setAttribute('nodename', 'asVec')
409
410 out = ng.addOutput(ng.createValidChildName('out'), 'color3')
411 out.setAttribute('nodename', 'asColor')
412
413 return graphDoc
