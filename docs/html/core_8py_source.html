17 def getBuiltinConfigs(self):
18 '''
19 Get the OCIO built in configurations.
20 Returnes a dictionary of color spaces and the default `ACES Cg Config`.
21 '''
22
23 # As of version 2.2, `ACES Cg Config` and `ACES Studio Config` are packaged with `OCIO`, meaning that
24 # they are available to use without having to download them separately. The `getBuiltinConfigs()`
25 # API is explained [here](https://opencolorio.readthedocs.io/en/latest/releases/ocio_2_2.html)
26
27 # Get the OCIO built in configs
28 registry = OCIO.BuiltinConfigRegistry().getBuiltinConfigs()
29
30 # Create a dictionary of configs
31 configs = {}
32 for item in registry:
33 # The short_name is the URI-style name.
34 # The ui_name is the name to use in a user interface.
35 short_name, ui_name, isRecommended, isDefault = item
36
37 # Don't present built-in configs to users if they are no longer recommended.
38 if isRecommended:
39 # Create a config using the Cg config
40 config = OCIO.Config.CreateFromBuiltinConfig(short_name)
41 colorSpaces = None
42 if config:
43 colorSpaces = config.getColorSpaces()
44
45 if colorSpaces:
46 configs[short_name] = [config, colorSpaces]
47
48 acesCgConfigPath = 'ocio://cg-config-v1.0.0_aces-v1.3_ocio-v2.1'
49 builtinCfgC = OCIO.Config.CreateFromFile(acesCgConfigPath)
50 print('Built-in config:', builtinCfgC.getName())
51 csnames = builtinCfgC.getColorSpaceNames()
52 print('- Number of color spaces: %d' % len(csnames))
53
54 return configs, builtinCfgC

84 def generateShaderCode(self, config, sourceColorSpace, destColorSpace, language):
85 '''
86 Generate shader for a transform from a source color space to a destination color space
87 for a given config and shader language.
88
89 Returns the shader code and the number of texture resources required.
90 '''
91 shaderCode = ''
92 textureCount = 0
93 if not config:
94 return shaderCode, textureCount
95
96 # Create a processor for a pair of colorspaces
97 processor = None
98 try:
99 processor = config.getProcessor(sourceColorSpace, destColorSpace)
100 except:
101 print('Failed to generated code for transform: %s -> %s' % (sourceColorSpace, destColorSpace))
102 return shaderCode, textureCount
103
104 if processor:
105 gpuProcessor = processor.getDefaultGPUProcessor()
106 if gpuProcessor:
107 shaderDesc = OCIO.GpuShaderDesc.CreateShaderDesc()
108 if shaderDesc:
109 try:
110 shaderDesc.setLanguage(language)
111 if shaderDesc.getLanguage() == language:
112 self.setShaderDescriptionParameters(shaderDesc, sourceColorSpace, destColorSpace, "color4")
113 gpuProcessor.extractGpuShaderInfo(shaderDesc)
114 shaderCode = shaderDesc.getShaderText()
115
116 for t in shaderDesc.getTextures():
117 textureCount += 1
118
119 if shaderCode:
120 shaderCode = shaderCode.replace(
121 "// Declaration of the OCIO shader function\n",
122 "// " + sourceColorSpace + " to " + destColorSpace + " function. Texture count: %d\n" % textureCount)
123
124 except OCIO.Exception as err:
125 print(err)
126
127 return shaderCode, textureCount

129 def hasTextureResources(self, configs, targetColorSpace, language):
130 '''
131 Scan through all the color spaces on the configs to check for texture resource usage.
132 Returns a set of color spaces that require texture resources.
133 '''
134 testedSources = set()
135 textureSources = set()
136 for c in configs:
137 config = OCIO.Config.CreateFromBuiltinConfig(c)
138 colorSpaces = config.getColorSpaces()
139 for colorSpace in colorSpaces:
140 colorSpaceName = colorSpace.getName()
141 # Skip if the colorspace is already tested
142 if colorSpaceName in testedSources:
143 continue
144 testedSources.add(colorSpaceName)
145
146 # Test for texture resource usage
147 code, textureCount = self.generateShaderCode(config, colorSpace.getName(), targetColorSpace, language)
148 if textureCount:
149 print('- Transform "%s" to "%s" requires %d texture resources' % (colorSpace.getName(), targetColorSpace, textureCount))
150 textureSources.add(colorSpaceName)
151
152 return textureSources

161 def OSL(self, config, sourceColorSpace, targetColorSpace):
162 if OCIO.GpuLanguage.LANGUAGE_OSL_1:
163 language = OCIO.GpuLanguage.LANGUAGE_OSL_1
164 code, textureCount = self.generateShaderCode(config, sourceColorSpace, targetColorSpace, language)
165 if code:
166 # Bit of ugly patching to make the main function name consistent.
167 transformName = self.createTransformName(sourceColorSpace, targetColorSpace, 'color4')
168 code = code.replace('OSL_' + transformName, '__temp_name__')
169 code = code.replace(transformName, transformName + '_impl')
170 code = code.replace('__temp_name__', transformName)
171 code = code.replace("// Declaration of the OCIO shader function\n", "// " + sourceColorSpace + " to " + targetColorSpace + " function\n")
172 code = '```c++\n' + code + '\n```\n'

174 def generateMaterialXDefinition(self, doc, sourceColorSpace, targetColorSpace, inputName, type):
175 '''
176 Create a new definition in a document for a given color space transform.
177 Returns the definition.
178 '''
179 # Create a definition
180 transformName = self.createTransformName(sourceColorSpace, targetColorSpace, type)
181 nodeName = transformName.replace('mx_', 'ND_')
182
183 comment = doc.addChildOfCategory('comment')
184 docString = ' Color space %s to %s transform. Generated via OCIO. ' % (sourceColorSpace, targetColorSpace)
185 comment.setDocString(docString)
186
187 definition = doc.addNodeDef(nodeName, 'color4')
188 category = sourceColorSpace + '_to_' + targetColorSpace
189 definition.setNodeString(category)
190 definition.setNodeGroup('colortransform')
191 definition.setDocString(docString)
192 definition.setVersionString('1.0')
193
194 defaultValueString = '0.0 0.0 0.0 1.0'
195 defaultValue = mx.createValueFromStrings(defaultValueString, 'color4')
196 input = definition.addInput(inputName, type)
197 input.setValue(defaultValue)
198 output = definition.getOutput('out')
199 output.setAttribute('default', 'in')
200
201 return definition

214 def createMaterialXImplementation(self, sourceColorSpace, targetColorSpace, doc, definition, transformName, extension, target):
215 '''
216 Create a new implementation in a document for a given definition.
217 '''
218 implName = transformName + '_' + target
219 filename = transformName + '.' + extension
220 implName = implName.replace('mx_', 'IM_')
221
222 # Check if implementation already exists
223 impl = doc.getImplementation(implName)
224 if impl:
225 print('Implementation already exists: %s' % implName)
226 return impl
227
228 comment = doc.addChildOfCategory('comment')
229 comment.setDocString(' Color space %s to %s transform. Generated via OCIO for target: %s'
230 % (sourceColorSpace, targetColorSpace, target))
231 impl = doc.addImplementation(implName)
232 impl.setFile(filename)
233 impl.setFunction(transformName)
234 impl.setTarget(target)
235 impl.setNodeDef(definition)
236
237 return impl

239 def generateOCIO(self, config, definitionDoc, implDoc, sourceColorSpace = 'acescg', targetColorSpace = 'lin_rec709',
240 type='color4', IN_PIXEL_STRING = 'in'):
241 '''
242 Generate a MaterialX definition and implementation for a given color space transform.
243 Returns the definition, implementation, source code, extension and target.
244 '''
245
246 # List of MaterialX target language, source code extensions, and OCIO GPU languages
247 generationList = [
248 ['genglsl', 'glsl', OCIO.GpuLanguage.GPU_LANGUAGE_GLSL_4_0]
249 #, ['genmsl', 'metal', OCIO.GpuLanguage.GPU_LANGUAGE_MSL_2_0]
250 ]
251
252 definition = None
253 transformName = self.createTransformName(sourceColorSpace, targetColorSpace, type)
254 for gen in generationList:
255 target = gen[0]
256 extension = gen[1]
257 language = gen[2]
258
259 code, textureCount = self.generateShaderCode(config, sourceColorSpace, targetColorSpace, language)
260
261 # Skip if there are texture resources
262 if textureCount:
263 print('- Skip generation for transform: "%s" to "%s" which requires %d texture resources' % (sourceColorSpace, targetColorSpace, textureCount))
264 continue
265
266 if code:
267 # Create the definition once
268 if not definition:
269 # Create color4 variant
270 definition = self.generateMaterialXDefinition(definitionDoc, sourceColorSpace, targetColorSpace,
271 IN_PIXEL_STRING, type)
272 # Create color3 variant (nodegraph)
273 self.createColor3Variant(definition, definitionDoc, IN_PIXEL_STRING)
274
275 # Create the implementation
276 self.createMaterialXImplementation(sourceColorSpace, targetColorSpace, implDoc, definition, transformName, extension, target)
277
278 return definition, transformName, code, extension, target

299 c4to3 = ng.addNode('convert', 'c4to3', 'color3')

307 ngout.setNodeName(c4to3.getName())
