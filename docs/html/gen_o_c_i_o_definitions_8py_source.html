1#!/usr/bin/env python
2'''
3
4The script will generate MaterialX definitions for all color spaces found in the
5ACES Cg Config` and `ACES Studio Config` configurations. The definitions will be
6generated for the following MaterialX targets:
7 - GLSL
8
9The output script will is:
101. A markdown file with information about the built-in configurations.
112. A color4 implementation file containing the source code generated by OCIO.
123. A color3 node graph implementation which uses the color4 implementation (nodegraph)
134. A file with both color3 and color4 MaterialX definitions (nodedef).
145. A MaterialX file containing implementation declarations for color3 and color4 variants.
15'''
16
17import os, argparse
18import MaterialX as mx
19import PyOpenColorIO as OCIO
20import core as mxocio
21
22def main():
23 parser = argparse.ArgumentParser(description="Create Materialx definitions using OCIO.")
24 parser.add_argument('--outputPath', dest='outputPath', help='File path to output material files to.')
25
26 opts = parser.parse_args()
27 outputPath = mx.FilePath("./data/")
28 if opts.outputPath:
29 outputPath = mx.FilePath(opts.outputPath)
30
31 # Check OCIO version
32 ver = OCIO.GetVersion()
33 ocioVersion = ver.split('.')
34 if len(ocioVersion) < 2:
35 print('OCIO version is not in the expected format.')
36 return
37 if int(ocioVersion[0]) < 2 or int(ocioVersion[1]) < 2:
38 print('OCIO version 2.2 or greater is required.')
39 return
40
41 print('OCIO version:', ver)
42 print('MaterialX version:', mx.getVersionString())
43
44 # Get the OCIO built in configs and write out the configuration information
45 # to a markdown file.
46 generator = mxocio.OCIOMaterialaxGenerator()
47 configs, aconfig = generator.getBuiltinConfigs()
48 md = generator.printConfigs(configs)
49 # Save configuration information as markdown
50 if not os.path.exists(outputPath.asString()):
51 os.makedirs(outputPath.asString())
52 configInfoFile = outputPath / mx.FilePath('OCIO_configurations.md')
53 print('Write out OCIO configurations to: ' + configInfoFile.asString())
54 f = open(configInfoFile.asString(), 'w')
55 f.write(md)
56
57 sourceColorSpace = "acescg"
58 targetColorSpace = 'lin_rec709'
59
60 # All code has the same input name
61 # It is possible to use a different name than the name used in the generated function ('inPixel')
62 IN_PIXEL_STRING = 'in'
63
64 # Generate MaterialX definitions and implementations for all color spaces
65 # found in the ACES Cg Config and ACES Studio Config configurations.
66 for c in configs:
67 config = configs[c][0]
68 for colorSpace in config.getColorSpaces():
69 aliases = colorSpace.getAliases()
70 trySource = ''
71 for alias in aliases:
72 # Get alias if it does not contain a space
73 if ' ' not in alias:
74 trySource = alias
75 if not trySource:
76 trySource = colorSpace.getName()
77 if trySource:
78 sourceColorSpace = trySource
79
80 # Skip if the source and target are the same
81 if sourceColorSpace == targetColorSpace:
82 continue
83
84 print('--- Generate transform for source color space:', trySource, '---')
85
86 definitionDoc = mx.createDocument()
87 implDoc = mx.createDocument()
88
89 definition, transformName, code, extension, target = generator.generateOCIO(aconfig, definitionDoc, implDoc, sourceColorSpace, targetColorSpace, 'color4')
90
91 # Write the definition, implementation and source code files
92 if definition:
93
94 filename = outputPath / mx.FilePath(definition.getName() + '.' + 'mtlx')
95 print('Write MaterialX definition file:', filename.asString())
96 mx.writeToXmlFile(definitionDoc, filename)
97
98 # Write the implementation document
99 implFileName = outputPath / mx.FilePath('IM_' + transformName + '.' + 'mtlx')
100 print('Write MaterialX implementation file:', implFileName.asString())
101 result = mx.writeToXmlFile(implDoc, implFileName)
102
103 generator.writeShaderCode(outputPath, code, transformName, extension, target)
104
105 else:
106 print('Could not find suitable color space name to use: ', colorSpace.getName())
107
108
109if __name__ == '__main__':
110 main()
